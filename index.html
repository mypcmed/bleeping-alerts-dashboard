Run GitHub Get Alerts JSON before Build Dashboard JSON.'); } function safeDate(value) { const d = new Date(value || now); return Number.isNaN(d.getTime()) ? now : d; } function normalizeSeverity(value) { const severity = String(value || 'low').toLowerCase();
